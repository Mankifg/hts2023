Janez se je spoprijateljil z roboti in je sestavil svoj kompliciran vpisni sistem, ki je tako zahteven, da ga roboti rešijo brez problemov medtem ko so ljudje prepočasni! Poznaš kakšnega robota, ki bi ti pomagal? 

from datetime import datetime
import random

def genflags():
    seed = datetime.utcnow().timestamp()
    seed = int(seed) // 5
    seed = seed ^ 0xDEADBEEF
    random.seed(seed)
    flags = [random.randint(0, 65535) for _ in range(8)]
    return flags


def testflags(data):
    flags = genflags()
    for i in range(1, 8):
        flags[i] = flags[i] ^ flags[i - 1]
    for i in range(8):
        if flags[i] != data[i]:
            return False
    return True
